datetime

from datetime import date, time, datetime

date(year, month, day)


>>> from datetime import date

>>> d = date(2013, 7, 9)

>>> d.year
2013
>>> d.month
7
>>> d.day
9

Comparisons


>>> today = date(2013, 7, 9)
>>> yesterday = date(2013, 7, 8)

>>> today > yesterday
True

Equality


>>> today = date(2013, 7, 9)
>>> also_today = date(2013, 7, 9)

>>> today == also_today
True

time(hour, min, sec, microsec)


>>> from datetime import time

>>> t = time(18, 30)

>>> t.hour
18
>>> t.minute
30
>>> t.second
0

Comparisons

Just like date()


>>> time(18, 0) > time(19, 0)
False
